CREATE TABLE INGREDIENT (
  NAME VARCHAR(255),
  LACTOSE_FREE BOOLEAN,
  VEGETARIAN BOOLEAN,
  VEGAN BOOLEAN,
  GLUTEN_FREE BOOLEAN
);
